		Pattern patronLetras = Pattern.compile("^[a-zA-Z ]*$");
		Pattern patronNumeros = Pattern.compile("^[0-9]*$");
		Pattern patronEmail = Pattern
				.compile("^[_a-z0-9-]+(\\.[_a-z0-9-]+)*@" + "[a-z0-9-]+(\\.[a-z0-9-]+)*(\\.[a-z]{2,4})$");

		Matcher mDni = patronNumeros.matcher(this.ventanaAlumnoAM.getDNI().getText());
		Matcher mNombre = patronLetras.matcher(this.ventanaAlumnoAM.getNombre().getText());
		Matcher mApellido = patronLetras.matcher(this.ventanaAlumnoAM.getApellido().getText());
		Matcher mTelefono = patronNumeros.matcher(this.ventanaAlumnoAM.getTelefono().getText());
		Matcher mEmail = patronEmail.matcher(this.ventanaAlumnoAM.getEmail().getText());

		String errores = " ";
		// Validaciones de null, y de tipos de datos validos
		if (!mDni.matches() || this.ventanaAlumnoAM.getDNI().getText().isEmpty()) {
			errores += " - Por favor, ingrese un DNI valido\n";
		}
		if (!mNombre.matches() || this.ventanaAlumnoAM.getNombre().getText().isEmpty()) {
			errores += " - Por favor, ingrese un nombre valido\n";
		}
		if (!mApellido.matches() || this.ventanaAlumnoAM.getApellido().getText().isEmpty()) {
			errores += " - Por favor, ingrese un apellido valido\n";
		}
		if (!mTelefono.matches()) {
			errores += " - Por favor, ingrese un telefono valido\n";
		}
		if (!mEmail.matches() || this.ventanaAlumnoAM.getEmail().getText().isEmpty()) {
			errores += " - Por favor, ingrese un email valido\n";
		}
		if(AlumnoManager.existePorDNI(this.ventanaAlumnoAM.getDNI().getText())){
			errores += " - Por favor, ingrese un dni nuevo. El dni ingresado ya existe\n";
		}
		if (errores.length() != 1) {
			JOptionPane.showMessageDialog(null, errores);
			return false;
		}
